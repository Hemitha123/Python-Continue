--order by 
select * from employee where salary=40000	--if salary equal to 40000
select * from employee where salary between 40000 and 50000		--between 40000 to 50000
select * from employee where empname like 'a%'		--only names staring with a 
select * from employee where empname like '%a'		--only the name ends with a
select * from employee where empname like '%a%' and gender='Female'   --the name swith a in name 
select * from employee order by designation		--sorts the designation is ascending order
select * from employee order by designation desc	----sorts the designation is decending order
select * from employee order by empname,place		--sorts the employee and place with priorty is ascending order
select * from employee where salary=50000 order by empname		--salary with 50000 employee name is sorted
--sum,min,max,avg,count
select sum(salary) from employee		--provides the sum of salary

select count(*) from employee where salary =50000   --counts the employee with salary 50000
